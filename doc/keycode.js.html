/**
 * @desc 根据keycode获得键名
 * @param  {number} keycode - code
 * @return {string} 返回的对应的key
 * @example
 * getKeyName(13);
 * // returns 'Enter'
 */

function getKeyName(keycode) {
  if (keyCodeMap[keycode]) {
    return keyCodeMap[keycode];
  } else {
    console.log("Unknow Key(Key Code:" + keycode + ")");
    return "";
  }
}
